    <script>
        // --- THREE.JS BACKGROUND ---
        let scene, camera, renderer, hearts = [];
        
        function initThree() {
            scene = new THREE.Scene();
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            renderer = new THREE.WebGLRenderer({ canvas: document.querySelector('#three-canvas'), alpha: true, antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);

            // Heart Shape Geometry
            const x = 0, y = 0;
            const heartShape = new THREE.Shape();
            heartShape.moveTo( x + 5, y + 5 );
            heartShape.bezierCurveTo( x + 5, y + 5, x + 4, y, x, y );
            heartShape.bezierCurveTo( x - 6, y, x - 6, y + 7,x - 6, y + 7 );
            heartShape.bezierCurveTo( x - 6, y + 11, x - 3, y + 15.4, x + 5, y + 19 );
            heartShape.bezierCurveTo( x + 12, y + 15.4, x + 16, y + 11, x + 16, y + 7 );
            heartShape.bezierCurveTo( x + 16, y + 7, x + 16, y, x + 10, y );
            heartShape.bezierCurveTo( x + 7, y, x + 5, y + 5, x + 5, y + 5 );

            const extrudeSettings = { depth: 2, bevelEnabled: true, bevelSegments: 2, steps: 2, bevelSize: 1, bevelThickness: 1 };
            const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);

            for (let i = 0; i < 25; i++) {
                const material = new THREE.MeshPhongMaterial({ 
                    color: i % 2 === 0 ? 0xec4899 : 0xef4444,
                    shininess: 100,
                    transparent: true,
                    opacity: 0.8
                });
                const mesh = new THREE.Mesh(geometry, material);
                
                // Random positioning
                mesh.position.set(
                    (Math.random() - 0.5) * 30,
                    (Math.random() - 0.5) * 30,
                    (Math.random() - 0.5) * 10
                );
                
                mesh.rotation.x = Math.PI; // Flip heart
                mesh.rotation.z = Math.random() * Math.PI;
                const s = Math.random() * 0.05 + 0.02;
                mesh.scale.set(s, s, s);
                
                mesh.userData = {
                    ySpeed: Math.random() * 0.01 + 0.005,
                    rotSpeed: Math.random() * 0.02,
                    offset: Math.random() * 100
                };
                
                scene.add(mesh);
                hearts.push(mesh);
            }

            const light = new THREE.DirectionalLight(0xffffff, 1.5);
            light.position.set(1, 1, 5);
            scene.add(light);
            scene.add(new THREE.AmbientLight(0xffffff, 0.5));

            camera.position.z = 15;
        }

        function animate() {
            requestAnimationFrame(animate);
            hearts.forEach(h => {
                h.position.y += Math.sin(Date.now() * 0.001 + h.userData.offset) * 0.005;
                h.position.x += Math.cos(Date.now() * 0.001 + h.userData.offset) * 0.002;
                h.rotation.y += h.userData.rotSpeed;
            });
            renderer.render(scene, camera);
        }

        // --- NAVIGATION ---
        let currentStep = 1;
        function navigate(step) {
            const outEl = document.getElementById(`step-${currentStep}`);
            const inEl = document.getElementById(`step-${step}`);
            
            // GSAP Transition
            gsap.to(outEl, { 
                opacity: 0, 
                y: -50, 
                scale: 0.9, 
                duration: 0.6, 
                ease: "power2.inOut",
                onComplete: () => {
                    outEl.classList.remove('active-card');
                    inEl.classList.add('active-card');
                    
                    gsap.fromTo(inEl, 
                        { opacity: 0, y: 50, scale: 0.9 }, 
                        { opacity: 1, y: 0, scale: 1, duration: 0.8, ease: "back.out(1.5)" }
                    );

                    // Stagger child elements
                    gsap.from(inEl.querySelectorAll('h1, p, button, .bento-item, .polaroid-wrapper'), {
                        opacity: 0,
                        y: 20,
                        stagger: 0.1,
                        duration: 0.5,
                        delay: 0.2
                    });
                }
            });

            currentStep = step;
            document.getElementById('progress-fill').style.width = `${(step / 5) * 100}%`;
        }

        // --- POLAROID PARALLAX ---
        const polaroid = document.getElementById('polaroid');
        document.addEventListener('mousemove', (e) => {
            if (currentStep !== 4) return;
            const x = (window.innerWidth / 2 - e.pageX) / 25;
            const y = (window.innerHeight / 2 - e.pageY) / 25;
            polaroid.style.transform = `rotateY(${x}deg) rotateX(${y}deg) rotateZ(-3deg)`;
        });

        // --- FINALE ---
        function startFinale() {
            const btn = document.getElementById('celebrate-btn');
            const finalWish = document.getElementById('final-wish');
            
            btn.disabled = true;
            gsap.to(btn, { opacity: 0, y: 20, duration: 0.5 });
            
            setTimeout(() => {
                finalWish.classList.remove('hidden');
                gsap.from(finalWish, { opacity: 0, y: 30, duration: 1, ease: "power3.out" });
            }, 500);

            // Confetti Sequence
            const duration = 5 * 1000;
            const animationEnd = Date.now() + duration;

            const randomInRange = (min, max) => Math.random() * (max - min) + min;

            const interval = setInterval(function() {
                const timeLeft = animationEnd - Date.now();

                if (timeLeft <= 0) return clearInterval(interval);

                const particleCount = 50 * (timeLeft / duration);
                
                // Burst from corners
                confetti({
                    particleCount,
                    startVelocity: 30,
                    spread: 360,
                    origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 },
                    colors: ['#ec4899', '#ef4444', '#ffd700']
                });
                confetti({
                    particleCount,
                    startVelocity: 30,
                    spread: 360,
                    origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 },
                    colors: ['#ec4899', '#ef4444', '#ffd700']
                });
                
                // Emojis
                confetti({
                    particleCount: 2,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0 },
                    shapes: ['text'],
                    shapeOptions: { text: { value: ['❤️', '✨', '💖'] } }
                });
                confetti({
                    particleCount: 2,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1 },
                    shapes: ['text'],
                    shapeOptions: { text: { value: ['❤️', '✨', '💖'] } }
                });
            }, 250);

            // Three.js hearts finale: swirl and float away
            hearts.forEach((h, i) => {
                gsap.to(h.position, {
                    x: h.position.x * 5,
                    y: 50,
                    z: 20,
                    duration: 4,
                    delay: i * 0.05,
                    ease: "power1.in"
                });
                gsap.to(h.material, { opacity: 0, duration: 3, delay: 1 });
            });
        }

        // Window resize
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        // Init
        initThree();
        animate();
    </script>
